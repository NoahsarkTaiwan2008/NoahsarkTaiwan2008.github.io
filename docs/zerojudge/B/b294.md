# b294. 經濟大恐荒
---
```java
import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        int N = scanner.nextInt();
        int ans = 0;
        for (int i = 1; i <= N; i++) {
            int tmp = scanner.nextInt();
            ans += tmp * i;
        }
        System.out.println(ans);
        scanner.close();
    }
}
```
